Module Option.Syntax

Binding operators. See manual section 12.23 for details.

  • since 5.5
let (let*): option('a) => ('a => option('b)) => option('b);

( let* ) is Option.bind.

let (and*): option('a) => option('b) => option(('a, 'b));

( and* ) is Option.product.

let (let+): option('a) => ('a => 'b) => option('b);

( let+ ) is Option.map.

let (and+): option('a) => option('b) => option(('a, 'b));

( and+ ) is Option.product.
